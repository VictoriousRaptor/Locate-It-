locateit_find_resource = {
	if = {
		limit = { 
			has_event_chain = locateit_res_chain
		}
		end_event_chain = locateit_res_chain
	}
	begin_event_chain = {
		event_chain = locateit_res_chain
		target = root
	}
	# Assure that at least one SP can be researched
	capital_star = {
		enable_special_project = {
			name = locateit_resource_sp_capital
			location = this
			owner = root
		}
	}
	set_variable = {
		which = count
		value = 0
	}
	every_galaxy_planet = {
		limit = {
			has_resource = $RESOURCE$
			# Must be surveyed
			is_surveyed = {
				who = root
				status = yes
			}
		}
		enable_special_project = {
			name = locateit_resource_sp
			location = THIS
			owner = root
		}
		prev = {
			change_variable = {
				which = count
				value = 1
			}
		}
	}
	if = {
		limit = {
			check_variable = {
				which = count
				value = 0
			}
		}
		end_event_chain = locateit_res_chain
		country_event = {
			id = locateit.99
		}
	}
}

locateit_begin_chain = {
	if = {
		limit = { 
			has_event_chain = $CHAIN_NAME$
		}
		end_event_chain = $CHAIN_NAME$
	}
	begin_event_chain = {
		event_chain = $CHAIN_NAME$
		target = root
	}
	capital_star = {
		enable_special_project = {
			name = $CLEANER_PROJECT_NAME$
			owner = root
		}
	}
}
