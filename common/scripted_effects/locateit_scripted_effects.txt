locateit_find_resource = {
	if = {
		limit = { 
			has_event_chain = locateit_res_chain
		}
		end_event_chain = locateit_res_chain
	}
	begin_event_chain = {
		event_chain = locateit_res_chain
		target = root
	}
	# set_variable = {
	# 	which = sr_name
	# 	value = $RESOURCE$
	# }
	# log = "sr_name: [This.sr_name]"
	# log = "sr_name: [This.sr_name.GetName]"
	# Assure that at least one SP can be researched
	enable_special_project = {
		name = locateit_resource_sp_hide
		owner = root
	}
	every_galaxy_planet = {
		limit = {
			has_resource = $RESOURCE$
			# Must be surveyed
			is_surveyed = {
				who = root
				status = yes
			}
		}
		enable_special_project = {
			name = locateit_resource_sp
			location = THIS
			owner = root
		}
	}
}

locateit_mega_set_chain = {
	if = {
		limit = { 
			has_event_chain = $CHAIN_NAME$
		}
		end_event_chain = $CHAIN_NAME$
	}
	begin_event_chain = {
		event_chain = $CHAIN_NAME$
		target = root
	}
	enable_special_project = {
		name = $PROJECT_NAME$
		owner = root
	}
}
