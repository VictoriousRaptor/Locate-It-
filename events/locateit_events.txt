# Unexploited
country_event = {
	id = locateit.109
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = { 
				has_event_chain = locateit_res_chain
			}
			end_event_chain = locateit_res_chain
		}
		begin_event_chain = {
			event_chain = locateit_res_chain
			target = root
		}
		# Assure that at least one SP can be researched
		enable_special_project = {
			name = locateit_resource_sp_capital
			owner = root
		}
		every_planet_within_border = {
			limit = {
				has_research_station = no
				has_mining_station = no
				has_any_megastructure = no
				is_colony = no
				OR = {
					AND = {
						has_resource = volatile_motes
						root = {
							has_technology = tech_mine_volatile_motes
						}
					}
					AND = {
						has_resource = exotic_gases
						root = {
							has_technology = tech_mine_exotic_gases
						}
					}
					AND = {
						has_resource = rare_crystals
						root = {
							has_technology = tech_mine_rare_crystals
						}
					}
					AND = {
						has_resource = sr_dark_matter
						root = {
							has_technology = tech_mine_dark_matter
						}
					}
					AND = {
						has_resource = sr_zro
						root = {
							has_technology = tech_mine_zro
						}
					}
					AND = {
						has_resource = sr_living_metal
						root = {
							has_technology = tech_mine_living_metal
						}
					}
					has_resource = nanites
				}
			}
			enable_special_project = {
				name = locateit_resource_sp
				location = THIS
				owner = root
			}
		}
	}
}

# All
country_event = {
	id = locateit.110
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = { 
				has_event_chain = locateit_res_chain
			}
			end_event_chain = locateit_res_chain
		}
		begin_event_chain = {
			event_chain = locateit_res_chain
			target = root
		}
		enable_special_project = {
			name = locateit_resource_sp_capital
			owner = root
		}
		every_galaxy_planet = {
			limit = {
				OR = {
					has_resource = volatile_motes
					has_resource = exotic_gases
					has_resource = rare_crystals
					has_resource = sr_dark_matter
					has_resource = sr_zro
					has_resource = sr_living_metal
					has_resource = nanites
				}
				# Must be surveyed
				is_surveyed = {
					who = root
					status = yes
				}
			}
			enable_special_project = {
				name = locateit_resource_sp
				location = THIS
				owner = root
			}
		}
	}
}
